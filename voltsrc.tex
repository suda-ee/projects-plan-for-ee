\section{任务和要求}
\begin{itemize}
\item 设计并制作一台连续可调的正负直流稳压电源
\item 主要技术指标与要求~(对正或负各部分而言)
    \begin{itemize}
    \item 输出直流电压~$V_{\rm O}=3\sim \unit{12}{\volt}$
    \item 最大输出电流~$I_{\rm LM}=\unit{1}{\ampere}$
    \item 输出纹波电压小于~\unit{5}{\milli\volt}，稳压系数~$S_{\rm
    r}\leq 5\times 10^{-3}$，输出内阻小于~\unit{0.1}{\ohm}
    \end{itemize}
\end{itemize}
